#include<stdio.h>
int arr[10],n;

//Insert Function//
void arr_insert(int val, int idx){
    if(idx==n){
        arr[n] = val
  }
    for(int i=n; i>=idx;i--){
    arr[i]=arr[i-1];
    }
    arr[idx]=val;
    n++;
}
//Delete Function
void arr_delete(int idx){
    for(int i=idx; i<n-1;i++){
    arr[i]=arr[i+1];
    }
    n--;}
//Find Index Function
int index(int ele){
    for (int i=0;i<n; i++){
        if (arr[i] == ele){
            return i;
            break;
 }
    }
    return -1;
    }
//Print Function
void print(){
    int i;
    for(i=0;i<n;i++){
    printf("%d ",arr[i])
}
    printf("\n");
}
